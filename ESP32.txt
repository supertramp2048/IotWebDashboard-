/* DISTANCE - WATCHDOG SAFE FIX (GIỮ NGUYÊN TỪ CODE CŨ) */
float getDistance() {
  portENTER_CRITICAL(&mux);
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);
  portEXIT_CRITICAL(&mux);  
  
  long duration = pulseIn(ECHO_PIN, HIGH, 30000);
  
  if (duration == 0) return 999;
  return duration * 0.034 / 2.0;
}

/* LCD HELPER (Sử dụng Mutex chống nhiễu) */
void updateLCD(String line1, String line2) {
    if (xSemaphoreTake(i2cMutex, pdMS_TO_TICKS(100)) == pdTRUE) {
        while(line1.length() < 16) line1 += " ";
        while(line2.length() < 16) line2 += " ";
        lcd.setCursor(0, 0); lcd.print(line1);
        lcd.setCursor(0, 1); lcd.print(line2);
        xSemaphoreGive(i2cMutex);
    }
}

String toMaskedText(size_t len) {
  String masked = "";
  for (size_t i = 0; i < len; i++) masked += "*";
  return masked;
}

void savePasswordToEEPROM(const String &pass) {
  for (int i = 0; i < PASS_LEN; i++) {
    EEPROM.write(ADDR_PASS + i, pass[i]);
  }
  EEPROM.commit();
}

void loadPasswordFromEEPROM() {
  String loaded = "";
  bool valid = true;

  for (int i = 0; i < PASS_LEN; i++) {
    char c = (char)EEPROM.read(ADDR_PASS + i);
    if (c < '0' || c > '9') {
      valid = false;
      break;
    }
    loaded += c;
  }

  if (!valid || loaded.length() != PASS_LEN) {
    currentPassword = "1234";
    savePasswordToEEPROM(currentPassword);
    return;
  }

  currentPassword = loaded;
}

char readKeypadKey() {
  char key = 0;
  if (xSemaphoreTake(i2cMutex, pdMS_TO_TICKS(50)) == pdTRUE) {
    if (customKeypad.isPressed()) {
      uint8_t idx = customKeypad.getKey();
      if (idx < 16) key = keymap[idx];
    }
    xSemaphoreGive(i2cMutex);
  }
  return key;
}

bool collectPinInput(String &output, const String &line1, bool allowCancel = true) {
  output = "";
  updateLCD(line1, "D:Huy");

  while (output.length() < PASS_LEN) {
    char key = readKeypadKey();

    if (!key) {
      vTaskDelay(80 / portTICK_PERIOD_MS);
      continue;
    }

    if (key >= '0' && key <= '9') {
      output += key;
      updateLCD(line1, toMaskedText(output.length()));
    } else if (allowCancel && key == 'D') {
      return false;
    }

    vTaskDelay(180 / portTICK_PERIOD_MS);
  }

  return true;
}

    updateLCD("Da huy doi MK", "");
    vTaskDelay(800 / portTICK_PERIOD_MS);
    return;

}

/* MQTT RPC - FIX HEAP FRAGMENTATION (GIỮ NGUYÊN TỪ CODE CŨ) */
void mqttCallback(char *topic, byte *payload, unsigned int length) {
  char msg[512] = {0};
  int msgLen = (length < 511) ? length : 511;
  memcpy(msg, payload, msgLen);
  msg[msgLen] = '\0';
  
  Serial.println(msg);
  
  bool state = strstr(msg, "\"params\":true") != NULL;
  
  if (strstr(msg, "setLed1") != NULL) digitalWrite(LED_PINS[0], state);
  if (strstr(msg, "setLed2") != NULL) digitalWrite(LED_PINS[1], state);
  if (strstr(msg, "setLed3") != NULL) digitalWrite(LED_PINS[2], state);
  if (strstr(msg, "setLed4") != NULL) digitalWrite(LED_PINS[3], state);
  
  if (strstr(msg, "setWindow") != NULL && !blockAppControl) {
    windowOpen = state; 
  }
  
  if (strstr(msg, "setGarage") != NULL && !blockAppControl) {
    portENTER_CRITICAL(&garaStateMux);
    garageOpen = state;
    portEXIT_CRITICAL(&garaStateMux);
    Serial.print("garageOpen = ");
    Serial.println(garageOpen);
  }
  
  if (strstr(msg, "setCurtain") != NULL) {
    curtainClosed = state;
  }

  if (strstr(msg, "setAutoCurtain") != NULL) {
    autoCurtainEnabled = state;
  }

  // THÊM MỚI: Điều khiển cửa từ App
  if (strstr(msg, "setDoor") != NULL) {
      portENTER_CRITICAL(&doorStateMux);
      doorOpen = state;
      portEXIT_CRITICAL(&doorStateMux);
  }
}

/* WIFI TASK (GIỮ NGUYÊN TỪ CODE CŨ) */
void wifiTask(void *pv) {
  Serial.println("\n=== WiFi Task Started ===");
  Serial.print("Connecting to SSID: ");
  Serial.println(WIFI_SSID);

  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);

  int wifiAttempts = 0;
  while (WiFi.status() != WL_CONNECTED && wifiAttempts < 20) {
    Serial.print(".");
    vTaskDelay(500 / portTICK_PERIOD_MS);
    wifiAttempts++;
  }

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\n[OK] WiFi Connected!");
    Serial.print("IP Address: ");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println("\n[FAIL] WiFi Connection Failed!");
  }

  for (;;) {
    if (WiFi.status() != WL_CONNECTED) {
      Serial.println("[WARNING] WiFi Disconnected! Reconnecting...");
      WiFi.disconnect();
      vTaskDelay(1000 / portTICK_PERIOD_MS);
      WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
    }
    vTaskDelay(5000 / portTICK_PERIOD_MS);
  }
}

/* MQTT TASK (GIỮ NGUYÊN TỪ CODE CŨ) */
void mqttTask(void *pv) {
  Serial.println("\n=== MQTT Task Started ===");
  
  int waitCount = 0;
  while (WiFi.status() != WL_CONNECTED && waitCount < 30) {
    vTaskDelay(1000 / portTICK_PERIOD_MS);
    waitCount++;
  }

  if (WiFi.status() != WL_CONNECTED) {
    vTaskDelete(NULL);
    return;
  }

  client.setServer(TB_SERVER, 1883);
  client.setCallback(mqttCallback);

  int mqttAttempts = 0;
  for (;;) {
    if (!client.connected()) {
      mqttAttempts++;
      if (client.connect("ESP32_DEVICE", TB_TOKEN, NULL)) {
        Serial.println("[OK] MQTT Connected to ThingsBoard!");
        client.subscribe("v1/devices/me/rpc/request/+");
        mqttAttempts = 0; 
      } else {
        vTaskDelay(5000 / portTICK_PERIOD_MS);
      }
    } else {
      if (mqttAttempts > 0) mqttAttempts = 0;
    }
    client.loop();
    vTaskDelay(10 / portTICK_PERIOD_MS);
  }
}

/* SENSOR TASK (GIỮ NGUYÊN TỪ CODE CŨ) */
void sensorTask(void *pv) {
  Serial.println("Sensor Task");
  for (;;) {
    gasValue = analogRead(GAS_SENSOR_PIN);
    fireDetected = (digitalRead(FIRE_SENSOR_PIN) == LOW);
    pirState = digitalRead(PIR_PIN);
    lightState = digitalRead(LIGHT_SENSOR_PIN);
    distanceCm = getDistance();
    if (pirState) pirLastActive = millis();

    digitalWrite(LED_PIR_PIN, (millis() - pirLastActive < PIR_ON_TIME) ? HIGH : LOW);

    vTaskDelay(SENSOR_INTERVAL / portTICK_PERIOD_MS);
  }
}

/* EMERGENCY TASK (GIỮ NGUYÊN TỪ CODE CŨ) */
void emergencyTask(void *pv) {
  Serial.println("Emergency Task");
  for (;;) {
    if ((fireDetected || gasValue > 1000) && !emergencyMode) {
      emergencyMode = true;
      blockAppControl = true;
      windowOpen = true;
      
      portENTER_CRITICAL(&garaStateMux);
      garageOpen = true;
      portEXIT_CRITICAL(&garaStateMux);
      
      curtainClosed = true;
      tone(BUZZER_PIN, 3000);
    }

    if (emergencyMode && !fireDetected && gasValue <= 1000) {
      emergencyMode = false;
      blockAppControl = false;
      windowOpen = false;
      
      portENTER_CRITICAL(&garaStateMux);
      garageOpen = false;
      portEXIT_CRITICAL(&garaStateMux);
      
      curtainClosed = false;
      noTone(BUZZER_PIN);
    }
    vTaskDelay(EMERGENCY_INTERVAL / portTICK_PERIOD_MS);
  }
}

/* SERVO TASK - FIX JITTER & POWER CONSUMPTION (CẬP NHẬT THÊM DOOR SERVO) */
void servoTask(void *pv) {
  Serial.println("Servo Task");
  TickType_t lastWake = xTaskGetTickCount();
  
  int lastWindowAngle = -1;
  int lastGarageAngle = -1;
  int lastCurtainAngle = -1;
  int lastDoorAngle = -1; // THÊM MỚI

  for (;;) {
    bool garage;
    portENTER_CRITICAL(&garaStateMux);
    garage = garageOpen;
    portEXIT_CRITICAL(&garaStateMux);

    if (lightState != lastLightState) {
      lastLightState = lightState;
      lastLightChangeTime = millis();
    }

    if ((millis() - lastLightChangeTime) > LIGHT_DEBOUNCE) {
      if (confirmedLightState != lightState && !emergencyMode && autoCurtainEnabled) {
        confirmedLightState = lightState;
        curtainClosed = confirmedLightState;
      }
    }

    int newWindowAngle = windowOpen ? 90 : 0;
    int newGarageAngle = garage ? 180 : 0;
    int newCurtainAngle = curtainClosed ? 0 : 180;
    int newDoorAngle = doorOpen ? 180 : 0; // THÊM MỚI
    
    if (newWindowAngle != lastWindowAngle) {
      windowServo.write(newWindowAngle);
      lastWindowAngle = newWindowAngle;
    }
    
    if (newGarageAngle != lastGarageAngle) {
      garageServo.write(newGarageAngle);
      lastGarageAngle = newGarageAngle;
    }
    
    if (newCurtainAngle != lastCurtainAngle) {
      curtainServo.write(newCurtainAngle);
      lastCurtainAngle = newCurtainAngle;
    }

    // THÊM MỚI
    if (newDoorAngle != lastDoorAngle) {
      doorServo.write(newDoorAngle);
      lastDoorAngle = newDoorAngle;
    }

    if (!emergencyMode) {
      if (distanceCm < 5) tone(BUZZER_PIN, 2000);
      else noTone(BUZZER_PIN);
    }

    vTaskDelayUntil(&lastWake, SERVO_INTERVAL / portTICK_PERIOD_MS);
  }
}

/* SETUP */
void setup() {
  Serial.begin(115200);
  delay(2000);
  Serial.println("Boot ok");

  // KHỞI TẠO I2C
  i2cMutex = xSemaphoreCreateMutex();
  Wire.begin(I2C_SDA, I2C_SCL);
  Wire.setClock(100000); // Ổn định I2C
  lcd.init(); lcd.backlight();
  customKeypad.begin();

  EEPROM.begin(EEPROM_SIZE);
  loadPasswordFromEEPROM();

  for (int i = 0; i < 4; i++) {
    pinMode(LED_PINS[i], OUTPUT);
  }

  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);
  pinMode(BUZZER_PIN, OUTPUT);
  pinMode(PIR_PIN, INPUT);
  pinMode(LED_PIR_PIN, OUTPUT);
  pinMode(FIRE_SENSOR_PIN, INPUT);
  pinMode(LIGHT_SENSOR_PIN, INPUT);

  ESP32PWM::allocateTimer(0);
  ESP32PWM::allocateTimer(1);
  ESP32PWM::allocateTimer(2);
  ESP32PWM::allocateTimer(3); // Cần thêm 1 timer cho Servo thứ 4

  windowServo.attach(WINDOW_SERVO_PIN);
  garageServo.attach(GARAGE_SERVO_PIN);
  curtainServo.attach(CURTAIN_SERVO_PIN);
  doorServo.attach(DOOR_SERVO_PIN); // Khởi tạo Servo cửa

  xTaskCreatePinnedToCore(wifiTask, "wifi", 6000, NULL, 3, NULL, 0);
  xTaskCreatePinnedToCore(mqttTask, "mqtt", 6000, NULL, 3, NULL, 0);
  xTaskCreatePinnedToCore(telemetryTask, "telemetry", 6000, NULL, 1, NULL, 0);

  xTaskCreatePinnedToCore(sensorTask, "sensor", 6000, NULL, 3, NULL, 1);
  xTaskCreatePinnedToCore(emergencyTask, "emergency", 6000, NULL, 4, NULL, 1);
  xTaskCreatePinnedToCore(servoTask, "servo", 6000, NULL, 3, NULL, 1);
  xTaskCreatePinnedToCore(lcdKeypadTask, "lcd_kp", 4096, NULL, 2, NULL, 1); // Đã chuyển sang Core 1

  Serial.println("SYSTEM STARTED");
}

void loop() {
  vTaskDelay(10000 / portTICK_PERIOD_MS);
